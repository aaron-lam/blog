```python
class Solution(object):
	def maxProfit(self, prices):
		buyPrice = prices[0]
		res = float('-inf')
		for i in range(1, len(prices)):
			res = max(res, prices[i] - buyPrice)
			buyPrice = min(buyPrice, prices[i])
		return 0 if res < 0 else res
```
